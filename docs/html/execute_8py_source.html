16def _parse_command_to_args(command: str) -> list[str]:
17 """Parse command string into argument list for safe subprocess execution.
18
19 Uses shlex.split() with posix=True to properly remove quotes from arguments.
20 This works correctly on both Windows and Unix platforms.
21
22 Args:
23 command: Command string to parse
24
25 Returns:
26 List of command arguments
27
28 Security:
29 - Uses shlex.split instead of manual parsing
30 - posix=True removes quotes from arguments (correct behavior for subprocess)
31 - Results in argument list for subprocess.run (no shell=True)
32
33 Examples:
34 >>> _parse_command_to_args("python --version")
35 ['python', '--version']
36 >>> _parse_command_to_args('echo "hello world"')
37 ['echo', 'hello world']
38 >>> _parse_command_to_args('python -c "print(\'test\')"')
39 ['python', '-c', "print('test')"]
40 """
41 # Use posix=True on all platforms for consistent quote removal
42 # This is the correct behavior for subprocess.run() which expects
43 # arguments without surrounding quotes
44 try:
45 args = shlex.split(command, posix=True)
46 if not args:
47 raise ValueError("Command cannot be empty after parsing")
48 return args
49 except ValueError as e:
50 raise SanitizedError(
51 f"Failed to parse command: {sanitize_error_message(str(e))}",
52 original_error=e
53 )

80def _enforce_output_limit(output: str, max_bytes: int) -> tuple[str, bool]:
81 """Enforce output size limit to prevent memory exhaustion.
82
83 Args:
84 output: Output string to check
85 max_bytes: Maximum allowed bytes
86
87 Returns:
88 Tuple of (potentially truncated output, was_truncated)
89
90 Examples:
91 >>> _enforce_output_limit("short", 1000)
92 ('short', False)
93 >>> _enforce_output_limit("x" * 2000, 1000)
94 ('x' * 1000 + '\\n[Output truncated...]', True)
95 """
96 output_bytes = output.encode('utf-8', errors='replace')
97
98 if len(output_bytes) <= max_bytes:
99 return output, False
100
101 # Truncate to max size
102 truncated_bytes = output_bytes[:max_bytes]
103 truncated_str = truncated_bytes.decode('utf-8', errors='replace')
104
105 # Add truncation marker
106 truncated_str += "\n[Output truncated...]"
107
108 return truncated_str, True

111def _execute_subprocess(
112 args: list[str],
113 cwd: Path | None,
114 timeout_seconds: float,
115 capture_output: bool
116) -> tuple[str, str, int, bool]:
117 """Execute command via subprocess with security measures.
118
119 Args:
120 args: Command arguments list
121 cwd: Working directory or None
122 timeout_seconds: Timeout in seconds
123 capture_output: Whether to capture stdout/stderr
124
125 Returns:
126 Tuple of (stdout, stderr, exit_code, timed_out)
127
128 Raises:
129 SanitizedError: If command execution fails
130
131 Security:
132 - NEVER uses shell=True
133 - Enforces timeout
134 - Handles all exception types
135 - Sanitizes error messages
136 """
137 try:
138 result = subprocess.run(
139 args,
140 capture_output=capture_output,
141 text=True,
142 timeout=timeout_seconds,
143 cwd=str(cwd) if cwd else None,
144 shell=False, # CRITICAL: Never use shell=True
145 check=False # Don't raise on non-zero exit
146 )
147
148 stdout = result.stdout if result.stdout else ""
149 stderr = result.stderr if result.stderr else ""
150 exit_code = result.returncode
151
152 return stdout, stderr, exit_code, False
153
154 except subprocess.TimeoutExpired:
155 # Command exceeded timeout
156 return "", f"Command exceeded timeout of {timeout_seconds}s", -1, True
157
158 except FileNotFoundError:
159 # Command not found
160 raise SanitizedError(
161 f"Command not found: {args[0]}. Ensure the command is installed and in PATH.",
162 original_error=None
163 )
164
165 except PermissionError as e:
166 # Permission denied
167 raise SanitizedError(
168 f"Permission denied executing command: {sanitize_error_message(str(e))}",
169 original_error=e
170 )
171
172 except OSError as e:
173 # Other OS errors
174 raise SanitizedError(
175 f"OS error executing command: {sanitize_error_message(str(e))}",
176 original_error=e
177 )
178
179 except Exception as e:
180 # Unexpected errors
181 raise SanitizedError(
182 f"Unexpected error executing command: {type(e).__name__}",
183 original_error=e
184 )

195 """Execute a command safely with timeout and output limits.
196
197 This tool executes commands without shell injection vulnerabilities.
198 Commands are parsed into argument lists and executed directly via
199 subprocess without shell=True.
200
201 Args:
202 command: Command to execute (e.g., "python --version", "npm test")
203 working_directory: Working directory for execution (default: current)
204 timeout_ms: Timeout in milliseconds (default: 30000, max: 300000)
205 capture_output: Whether to capture stdout/stderr (default: True)
206 ctx: MCP context for logging (optional)
207
208 Returns:
209 ExecuteCommandOutput with stdout, stderr, exit code, and timing
210
211 Raises:
212 ValueError: If input validation fails
213 SanitizedError: If command execution fails
214
215 Security:
216 - No shell=True (prevents shell injection)
217 - Command parsed with shlex.split (safe parsing)
218 - Mandatory timeout enforcement
219 - Output size limits (prevents memory exhaustion)
220 - Path validation for working directory
221 - Sanitized error messages (no information leakage)
222
223 Examples:
224 >>> result = await execute_command("python --version")
225 >>> print(result.stdout) # "Python 3.11.5"
226 >>> print(result.exit_code) # 0
227
228 >>> result = await execute_command(
229 ... "npm test",
230 ... working_directory="./myproject",
231 ... timeout_ms=60000
232 ... )
233 """
234 start_time = time.time()
235
236 # Validate inputs via Pydantic
237 input_data = ExecuteCommandInput(
238 command=command,
239 working_directory=working_directory,
240 timeout_ms=timeout_ms,
241 capture_output=capture_output
242 )
243
244 if ctx:
245 await ctx.info(f"Executing command: {input_data.command}")
246
247 # Parse command into argument list (security critical)
248 args = _parse_command_to_args(input_data.command)
249
250 # Validate working directory if provided
251 cwd = _validate_working_directory(input_data.working_directory)
252
253 # Execute command with timeout
254 timeout_seconds = input_data.timeout_ms / 1000.0
255 stdout, stderr, exit_code, timed_out = _execute_subprocess(
256 args=args,
257 cwd=cwd,
258 timeout_seconds=timeout_seconds,
259 capture_output=input_data.capture_output
260 )
261
262 # Enforce output size limits
263 max_output = config.max_output_bytes
264 stdout, stdout_truncated = _enforce_output_limit(stdout, max_output)
265 stderr, stderr_truncated = _enforce_output_limit(stderr, max_output)
266 output_truncated = stdout_truncated or stderr_truncated
267
268 # Calculate execution time

270
271 if ctx:
272 await ctx.info(
273 f"Command completed: exit_code={exit_code}, "
274 f"time={execution_time_ms}ms, timed_out={timed_out}"
275 )
276
277 return ExecuteCommandOutput(
278 stdout=stdout,
279 stderr=stderr,
280 exit_code=exit_code,
281 execution_time_ms=execution_time_ms,
